# [3702 재귀] 파스칼의 삼각형 2

```c++
#include <iostream>
#include <stdio.h>
using namespace std;
const int LEN = 50;
int arr[LEN][LEN];

int pascal(int r, int c){
    if(r == 1 ||c == 1) {arr[r][c] = 1; return 1;} //(1,1), (x, 1), (1, y)
    if(!arr[r][c]) arr[r][c] = (pascal(r, c-1) + pascal(r-1, c)) % 100000000;
    return arr[r][c];
}
int main(){
    int r, c;
    scanf("%d %d", &r, &c);
    pascal(r, c);
    printf("%d", arr[r][c]);
    return 0;
}
```
